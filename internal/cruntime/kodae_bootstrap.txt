static inline kodae_str kodae_str_lit(const char *c) {
  if (!c) { return (kodae_str){NULL, 0}; }
  return (kodae_str){c, (int64_t)strlen(c)};
}

const char* kodae_str_to_cstr(kodae_str s) {
  if (!s.data) { return ""; }
  char* p = (char*)malloc((size_t)s.len + 1);
  if (!p) { exit(1); }
  if (s.len > 0) { memcpy(p, s.data, (size_t)s.len); }
  p[s.len] = 0;
  return p;
}

kodae_str kodae_str_concat(kodae_str a, kodae_str b) {
  int64_t n = a.len + b.len;
  if (n <= 0) { return (kodae_str){"", 0}; }
  char *p = (char *)malloc((size_t)n + 1);
  if (!p) {
    fprintf(stderr, "kodae: oom in str concat\n");
    exit(1);
  }
  if (a.len > 0) { memcpy(p, a.data, (size_t)a.len); }
  if (b.len > 0) { memcpy(p + a.len, b.data, (size_t)b.len); }
  p[n] = 0;
  return (kodae_str){p, n};
}

kodae_str kodae_int_to_str(int64_t v) {
  char buf[32];
  int n = snprintf(buf, sizeof buf, "%" PRId64, v);
  if (n < 0) { n = 0; }
  if (n >= (int)sizeof buf) { n = (int)sizeof buf - 1; }
  char *p = (char *)malloc((size_t)n + 1);
  if (!p) { exit(1); }
  memcpy(p, buf, (size_t)n);
  p[n] = 0;
  return (kodae_str){p, n};
}

kodae_str kodae_float_to_str(double v) {
  char buf[64];
  int n = snprintf(buf, sizeof buf, "%.17g", v);
  if (n < 0) { n = 0; }
  char *p = (char *)malloc((size_t)n + 1);
  if (!p) { exit(1); }
  memcpy(p, buf, (size_t)n);
  p[n] = 0;
  return (kodae_str){p, n};
}

kodae_str kodae_input(kodae_str prompt) {
  if (prompt.data && prompt.len > 0) {
    fwrite(prompt.data, 1, (size_t)prompt.len, stdout);
    fflush(stdout);
  }
  char line[4096];
  if (fgets(line, (int)sizeof line, stdin) == NULL) {
    return (kodae_str){NULL, 0};
  }
  int64_t n = 0;
  while (n < (int64_t)sizeof line - 1 && line[n] && line[n] != '\n' && line[n] != '\r') {
    n++;
  }
  char *p = (char *)malloc((size_t)(n < 0 ? 0 : n) + 1);
  if (!p) { exit(1); }
  if (n > 0) { memcpy(p, line, (size_t)n); }
  p[n] = 0;
  return (kodae_str){p, n};
}

int64_t kodae_input_int(kodae_str prompt) {
  kodae_str s = kodae_input(prompt);
  if (s.len == 0) return 0;
  return strtoll(s.data, NULL, 10);
}

double kodae_input_float(kodae_str prompt) {
  kodae_str s = kodae_input(prompt);
  if (s.len == 0) return 0.0;
  return strtod(s.data, NULL);
}

static int kodae_random_seeded;
int64_t kodae_random(int64_t lo, int64_t hi) {
  if (!kodae_random_seeded) { srand((unsigned)time(NULL)); kodae_random_seeded = 1; }
  if (hi < lo) { int64_t t = lo; lo = hi; hi = t; }
  int64_t span = hi - lo + 1;
  if (span <= 0) { return lo; }
  return lo + (int64_t)(rand() % span);
}

double kodae_random_float(double lo, double hi) {
  if (!kodae_random_seeded) { srand((unsigned)time(NULL)); kodae_random_seeded = 1; }
  double r = (double)rand() / (double)RAND_MAX;
  return lo + r * (hi - lo);
}

bool kodae_random_bool(void) {
  if (!kodae_random_seeded) { srand((unsigned)time(NULL)); kodae_random_seeded = 1; }
  return rand() % 2 == 0;
}

bool kodae_chance(int64_t pct) {
  if (!kodae_random_seeded) { srand((unsigned)time(NULL)); kodae_random_seeded = 1; }
  return (rand() % 100) < pct;
}

void* kodae_random_pick(kodae_list* l) {
  if (!l || l->len <= 0) return NULL;
  return kodae_list_at_ptr(l, kodae_random(0, l->len - 1));
}

int64_t kodae_clamp_int(int64_t v, int64_t lo, int64_t hi) {
  if (v < lo) return lo;
  if (v > hi) return hi;
  return v;
}

  if (v < lo) return lo;
  if (v > hi) return hi;
  return v;
}

double kodae_lerp(double a, double b, double t) {
  return a + t * (b - a);
}

double kodae_map(double v, double in_lo, double in_hi, double out_lo, double out_hi) {
  return out_lo + (v - in_lo) * (out_hi - out_lo) / (in_hi - in_lo);
}

double kodae_distance(double x1, double y1, double x2, double y2) {
  double dx = x2 - x1;
  double dy = y2 - y1;
  return sqrt(dx * dx + dy * dy);
}

double kodae_angle_to(double x1, double y1, double x2, double y2) {
  return atan2(y2 - y1, x2 - x1);
}

bool kodae_file_exists(kodae_str path) {
  const char* c = kodae_str_to_cstr(path);
  struct stat st;
  bool ok = (stat(c, &st) == 0 && (st.st_mode & S_IFREG));
  free((void*)c);
  return ok;
}

bool kodae_folder_exists(kodae_str path) {
  const char* c = kodae_str_to_cstr(path);
  struct stat st;
  bool ok = (stat(c, &st) == 0 && (st.st_mode & S_IFDIR));
  free((void*)c);
  return ok;
}

kodae_str kodae_read_file(kodae_str path) {
  const char* c = kodae_str_to_cstr(path);
  FILE* f = fopen(c, "rb");
  free((void*)c);
  if (!f) return (kodae_str){NULL, 0};
  fseek(f, 0, SEEK_END);
  long size = ftell(f);
  fseek(f, 0, SEEK_SET);
  char* buf = (char*)malloc(size + 1);
  if (!buf) { fclose(f); return (kodae_str){NULL, 0}; }
  fread(buf, 1, size, f);
  buf[size] = 0;
  fclose(f);
  return (kodae_str){buf, (int64_t)size};
}

void kodae_write_file(kodae_str path, kodae_str content) {
  const char* c = kodae_str_to_cstr(path);
  FILE* f = fopen(c, "wb");
  free((void*)c);
  if (!f) return;
  if (content.len > 0) fwrite(content.data, 1, content.len, f);
  fclose(f);
}

void kodae_append_file(kodae_str path, kodae_str content) {
  const char* c = kodae_str_to_cstr(path);
  FILE* f = fopen(c, "ab");
  free((void*)c);
  if (!f) return;
  if (content.len > 0) fwrite(content.data, 1, content.len, f);
  fclose(f);
}

void kodae_delete_file(kodae_str path) {

kodae_list kodae_list_new(int64_t elem_size) {
  kodae_list l;
  l.data = NULL;
  l.len = 0;
  l.cap = 0;
  l.elem_size = elem_size;
  return l;
}

static void kodae_list_reserve(kodae_list* l, int64_t need) {
  if (!l || need <= l->cap) { return; }
  int64_t nc = l->cap > 0 ? l->cap : 4;
  while (nc < need) { nc *= 2; }
  size_t bytes = (size_t)(nc * l->elem_size);
  uint8_t* p = (uint8_t*)realloc(l->data, bytes);
  if (!p) {
    fprintf(stderr, "kodae: oom in list reserve\n");
    exit(1);
  }
  l->data = p;
  l->cap = nc;
}

void kodae_list_push(kodae_list* l, const void* elem) {
  if (!l || !elem) { return; }
  kodae_list_reserve(l, l->len + 1);
  memcpy(l->data + (size_t)(l->len * l->elem_size), elem, (size_t)l->elem_size);
  l->len++;
}

void kodae_list_append(kodae_list* l, const kodae_list* other) {
  if (!l || !other) { return; }
  if (l->elem_size != other->elem_size) {
    fprintf(stderr, "kodae: append element-size mismatch\n");
    exit(1);
  }
  kodae_list_reserve(l, l->len + other->len);
  if (other->len > 0) {
    memcpy(l->data + (size_t)(l->len * l->elem_size), other->data, (size_t)(other->len * other->elem_size));
  }
  l->len += other->len;
}

void kodae_list_pop(kodae_list* l, void* out) {
  if (!l || !out || l->len <= 0) {
    fprintf(stderr, "kodae: pop from empty list\n");
    exit(1);
  }
  l->len--;
  memcpy(out, l->data + (size_t)(l->len * l->elem_size), (size_t)l->elem_size);
}

void kodae_list_remove_at(kodae_list* l, int64_t idx, void* out) {
  if (!l || !out || idx < 0 || idx >= l->len) {
    fprintf(stderr, "kodae: remove index out of range\n");
    exit(1);
  }
  memcpy(out, l->data + (size_t)(idx * l->elem_size), (size_t)l->elem_size);
  int64_t tail = l->len - idx - 1;
  if (tail > 0) {
    memmove(l->data + (size_t)(idx * l->elem_size),
            l->data + (size_t)((idx + 1) * l->elem_size),
            (size_t)(tail * l->elem_size));
  }
  l->len--;
}

void* kodae_list_at_ptr(kodae_list* l, int64_t idx) {
  if (!l || idx < 0 || idx >= l->len) {
    fprintf(stderr, "kodae: index out of range\n");
    exit(1);
  }
  return l->data + (size_t)(idx * l->elem_size);
}

kodae_str kodae_str_upper(kodae_str s) {
  if (s.len <= 0) return s;
  char* p = (char*)malloc(s.len + 1);
  for (int64_t i = 0; i < s.len; i++) {
    char c = s.data[i];
    if (c >= 'a' && c <= 'z') c -= 32;
    p[i] = c;
  }
  p[s.len] = 0;
  return (kodae_str){p, s.len};
}

kodae_str kodae_str_lower(kodae_str s) {
  if (s.len <= 0) return s;
  char* p = (char*)malloc(s.len + 1);
  for (int64_t i = 0; i < s.len; i++) {
    char c = s.data[i];
    if (c >= 'A' && c <= 'Z') c += 32;
    p[i] = c;
  }
  p[s.len] = 0;
  return (kodae_str){p, s.len};
}

kodae_str kodae_str_trim(kodae_str s) {
  if (s.len <= 0) return s;
  int64_t start = 0;
  while (start < s.len && (unsigned char)s.data[start] <= 32) start++;
  int64_t end = s.len - 1;
  while (end >= start && (unsigned char)s.data[end] <= 32) end--;
  int64_t n = end - start + 1;
  if (n <= 0) return (kodae_str){"", 0};
  char* p = (char*)malloc(n + 1);
  memcpy(p, s.data + start, n);
  p[n] = 0;
  return (kodae_str){p, n};
}

bool kodae_str_contains(kodae_str s, kodae_str sub) {
  if (sub.len == 0) return true;
  if (s.len < sub.len) return false;
  for (int64_t i = 0; i <= s.len - sub.len; i++) {
    if (memcmp(s.data + i, sub.data, sub.len) == 0) return true;
  }
  return false;
}

kodae_str kodae_str_replace(kodae_str s, kodae_str old_s, kodae_str new_s) {
  if (old_s.len == 0) return s;
  int64_t count = 0;
  for (int64_t i = 0; i <= s.len - old_s.len; ) {
    if (memcmp(s.data + i, old_s.data, old_s.len) == 0) {
      count++;
      i += old_s.len;
    } else {
      i++;
    }
  }
  if (count == 0) return s;
  int64_t new_len = s.len + count * (new_s.len - old_s.len);
  char* p = (char*)malloc(new_len + 1);
  char* dest = p;
  for (int64_t i = 0; i < s.len; ) {
    if (i <= s.len - old_s.len && memcmp(s.data + i, old_s.data, old_s.len) == 0) {
      memcpy(dest, new_s.data, new_s.len);
      dest += new_s.len;
      i += old_s.len;
    } else {
      *dest++ = s.data[i++];
    }
  }
  *dest = 0;
  return (kodae_str){p, new_len};
}

kodae_list kodae_str_split(kodae_str s, kodae_str delim) {
  kodae_list l = kodae_list_new(sizeof(kodae_str));
  if (s.len == 0) return l;
  if (delim.len == 0) {
    for (int64_t i = 0; i < s.len; i++) {
      kodae_str sub = {s.data + i, 1};
      kodae_list_push(&l, &sub);
    }
    return l;
  }
  int64_t start = 0;
  for (int64_t i = 0; i <= s.len - delim.len; ) {
    if (memcmp(s.data + i, delim.data, delim.len) == 0) {
      kodae_str sub = {s.data + start, i - start};
      kodae_list_push(&l, &sub);
      i += delim.len;
      start = i;
    } else {
      i++;
    }
  }
  kodae_str last = {s.data + start, s.len - start};
  kodae_list_push(&l, &last);
  return l;
}

kodae_str kodae_str_slice(kodae_str s, int64_t start, int64_t end) {
  if (start < 0) start = 0;
  if (end > s.len) end = s.len;
  if (start >= end) return (kodae_str){"", 0};
  int64_t n = end - start;
  char* p = (char*)malloc(n + 1);
  memcpy(p, s.data + start, n);
  p[n] = 0;
  return (kodae_str){p, n};
}

kodae_list kodae_list_files(kodae_str path) {
  kodae_list l = kodae_list_new(sizeof(kodae_str));
  const char* c = kodae_str_to_cstr(path);
#if defined(_WIN32)
  struct _finddata_t fd;
  char pattern[4096];
  snprintf(pattern, sizeof pattern, "%s/*", c);
  intptr_t h = _findfirst(pattern, &fd);
  if (h != -1) {
    do {
      if (strcmp(fd.name, ".") != 0 && strcmp(fd.name, "..") != 0) {
        kodae_str s = kodae_str_lit(strdup(fd.name));
        kodae_list_push(&l, &s);
      }
    } while (_findnext(h, &fd) == 0);
    _findclose(h);
  }
#else
  DIR* d = opendir(c);
  if (d) {
    struct dirent* de;
    while ((de = readdir(d))) {
      if (strcmp(de->d_name, ".") != 0 && strcmp(de->d_name, "..") != 0) {
        kodae_str s = kodae_str_lit(strdup(de->d_name));
        kodae_list_push(&l, &s);
      }
    }
    closedir(d);
  }
#endif
  free((void*)c);
  return l;
}

void kodae_list_shuffle(kodae_list* l) {
  if (!l || l->len <= 1) return;
  if (!kodae_random_seeded) { srand((unsigned)time(NULL)); kodae_random_seeded = 1; }
  for (int64_t i = l->len - 1; i > 0; i--) {
    int64_t j = (int64_t)rand() % (i + 1);
    uint8_t tmp[1024]; 
    size_t sz = (size_t)l->elem_size;
    memcpy(tmp, l->data + (size_t)(i * l->elem_size), sz);
    memcpy(l->data + (size_t)(i * l->elem_size), l->data + (size_t)(j * l->elem_size), sz);
    memcpy(l->data + (size_t)(j * l->elem_size), tmp, sz);
  }
}

void kodae_list_reverse(kodae_list* l) {
  if (!l || l->len <= 1) return;
  for (int64_t i = 0; i < l->len / 2; i++) {
    int64_t j = l->len - 1 - i;
    uint8_t tmp[1024];
    size_t sz = (size_t)l->elem_size;
    memcpy(tmp, l->data + (size_t)(i * l->elem_size), sz);
    memcpy(l->data + (size_t)(i * l->elem_size), l->data + (size_t)(j * l->elem_size), sz);
    memcpy(l->data + (size_t)(j * l->elem_size), tmp, sz);
  }
}

static int kodae_cmp_i64(const void* a, const void* b) {
  int64_t va = *(const int64_t*)a;
  int64_t vb = *(const int64_t*)b;
  return (va > vb) - (va < vb);
}
static int kodae_cmp_f64(const void* a, const void* b) {
  double va = *(const double*)a;
  double vb = *(const double*)b;
  return (va > vb) - (va < vb);
}
static int kodae_cmp_str(const void* a, const void* b) {
  kodae_str sa = *(const kodae_str*)a;
  kodae_str sb = *(const kodae_str*)b;
  int64_t min_len = sa.len < sb.len ? sa.len : sb.len;
  int res = memcmp(sa.data, sb.data, (size_t)min_len);
  if (res == 0) return (sa.len > sb.len) - (sa.len < sb.len);
  return res;
}

void kodae_list_sort(kodae_list* l) {
  if (!l || l->len <= 1) return;
  if (l->elem_size == 8) {
    /* Heuristic: assume int64 (or double, they sort similarly for positives) */
    qsort(l->data, (size_t)l->len, (size_t)l->elem_size, kodae_cmp_i64);
  } else if (l->elem_size == sizeof(kodae_str)) {
    qsort(l->data, (size_t)l->len, (size_t)l->elem_size, kodae_cmp_str);
  }
}

kodae_str kodae_str_reverse(kodae_str s) {
  if (s.len <= 1) return s;
  char* d = (char*)malloc((size_t)s.len);
  for (int64_t i = 0; i < s.len; i++) d[i] = s.data[s.len - 1 - i];
  return (kodae_str){d, s.len};
}

kodae_str kodae_str_repeat(kodae_str s, int64_t n) {
  if (n <= 0) return (kodae_str){"", 0};
  int64_t total = s.len * n;
  char* d = (char*)malloc((size_t)total);
  for (int64_t i = 0; i < n; i++) memcpy(d + (size_t)(i * s.len), s.data, (size_t)s.len);
  return (kodae_str){d, total};
}

bool kodae_str_starts(kodae_str s, kodae_str sub) {
  if (sub.len > s.len) return false;

}

bool kodae_str_ends(kodae_str s, kodae_str sub) {
  if (sub.len > s.len) return false;
  return memcmp(s.data + (size_t)(s.len - sub.len), sub.data, (size_t)sub.len) == 0;
}

uint32_t kodae_color_rgb(int r, int g, int b) {
  return (uint32_t)(0xFF000000 | ((b & 0xFF) << 16) | ((g & 0xFF) << 8) | (r & 0xFF));
}

uint32_t kodae_color_rgba(int r, int g, int b, int a) {
  return (uint32_t)(((a & 0xFF) << 24) | ((b & 0xFF) << 16) | ((g & 0xFF) << 8) | (r & 0xFF));
}

uint32_t kodae_color_lerp(uint32_t a, uint32_t b, double t) {
  if (t < 0) t = 0; if (t > 1) t = 1;
  int r1 = a & 0xFF, g1 = (a >> 8) & 0xFF, b1 = (a >> 16) & 0xFF, a1 = (a >> 24) & 0xFF;
  int r2 = b & 0xFF, g2 = (b >> 8) & 0xFF, b2 = (b >> 16) & 0xFF, a2 = (b >> 24) & 0xFF;
  return kodae_color_rgba((int)(r1 + (r2-r1)*t), (int)(g1 + (g2-g1)*t), (int)(b1 + (b2-b1)*t), (int)(a1 + (a2-a1)*t));
}

uint32_t kodae_color_fade(uint32_t c, double alpha) {
  if (alpha < 0) alpha = 0; if (alpha > 1) alpha = 1;
  return (c & 0x00FFFFFF) | (((uint32_t)((c >> 24) * alpha)) << 24);
}

kodae_res_str kodae_http_get(kodae_str url) {
  if (!url.data || url.len == 0) return (kodae_res_str){{NULL,0}, kodae_str_lit("empty url"), false};
  char cmd[4096];
  snprintf(cmd, sizeof cmd, "curl -sL \"%.*s\"", (int)url.len, url.data);
#if defined(_WIN32)
  FILE *f = _popen(cmd, "r");
#else
  FILE *f = popen(cmd, "r");
#endif
  if (!f) return (kodae_res_str){{NULL,0}, kodae_str_lit("failed to launch curl"), false};
  
  int cap = 4096;
  int len = 0;
  char *buf = (char *)malloc(cap);
  while (!feof(f)) {
    if (len + 1024 > cap) {
      cap *= 2;
      buf = (char *)realloc(buf, cap);
    }
    size_t r = fread(buf + len, 1, 1024, f);
    if (r > 0) len += r;
  }
  buf[len] = 0;
#if defined(_WIN32)
  int status = _pclose(f);
#else
  int status = pclose(f);
#endif
  if (status != 0) {
    free(buf);
    return (kodae_res_str){{NULL,0}, kodae_str_lit("http error"), false};
  }
  return (kodae_res_str){{buf, len}, {NULL, 0}, true};
}

kodae_res_str kodae_http_post(kodae_str url, kodae_str body) {
  if (!url.data || url.len == 0) return (kodae_res_str){{NULL,0}, kodae_str_lit("empty url"), false};
  char cmd[8192];
  // Note: escaping body for cmd line is tricky, we write it to a temp file and use --data-binary @temp
  kodae_str temp_path = kodae_str_lit(".kodae_temp_post_body");
  kodae_write_file(temp_path, body);
  snprintf(cmd, sizeof cmd, "curl -sL -X POST --data-binary @.kodae_temp_post_body \"%.*s\"", (int)url.len, url.data);
#if defined(_WIN32)
  FILE *f = _popen(cmd, "r");
#else
  FILE *f = popen(cmd, "r");
#endif
  if (!f) return (kodae_res_str){{NULL,0}, kodae_str_lit("failed to launch curl"), false};
  
  int cap = 4096;
  int len = 0;
  char *buf = (char *)malloc(cap);
  while (!feof(f)) {
    if (len + 1024 > cap) {
      cap *= 2;
      buf = (char *)realloc(buf, cap);
    }
    size_t r = fread(buf + len, 1, 1024, f);
    if (r > 0) len += r;
  }
  buf[len] = 0;
#if defined(_WIN32)
  int status = _pclose(f);
#else
  int status = pclose(f);
#endif
  kodae_delete_file(temp_path);
  if (status != 0) {
    free(buf);
    return (kodae_res_str){{NULL,0}, kodae_str_lit("http error"), false};
  }
  return (kodae_res_str){{buf, len}, {NULL, 0}, true};
}

bool kodae_download(kodae_str url, kodae_str dest) {
  if (!url.data || url.len == 0 || !dest.data || dest.len == 0) return false;
  char cmd[4096];
  snprintf(cmd, sizeof cmd, "curl -sL -o \"%.*s\" \"%.*s\"", (int)dest.len, dest.data, (int)url.len, url.data);
  int status = system(cmd);
  return status == 0;
}

// JSON APIs

kodae_any kodae_json_parse(kodae_str s) {
    if (!s.data || s.len == 0) return NULL;
    char* cstr = (char*)kodae_str_to_cstr(s);
    kodae_any a = json_parse_string(cstr);
    free(cstr);
    return a;
}

kodae_str kodae_str_new(const char* s) {
    if (!s) return (kodae_str){NULL, 0};
    int64_t n = (int64_t)strlen(s);
    if (n == 0) return (kodae_str){"", 0};
    char* p = (char*)malloc((size_t)n + 1);
    if (!p) exit(1);
    memcpy(p, s, (size_t)n);
    p[n] = 0;
    return (kodae_str){p, n};
}

kodae_str kodae_json_build(kodae_any a) {
    if (!a) return (kodae_str){NULL, 0};
    char* s = json_serialize_to_string(a);
    if (!s) return (kodae_str){NULL, 0};
    kodae_str res = kodae_str_new(s);
    json_free_serialized_string(s);
    return res;
}

kodae_any kodae_any_get(kodae_any a, kodae_str key) {
    if (!a || json_value_get_type(a) != JSONObject) return NULL;
    char* ckey = (char*)kodae_str_to_cstr(key);
    kodae_any v = json_object_get_value(json_value_get_object(a), ckey);
    free(ckey);
    return v;
}

kodae_any kodae_any_at(kodae_any a, int64_t idx) {
    if (!a || json_value_get_type(a) != JSONArray) return NULL;
    return json_array_get_value(json_value_get_array(a), (size_t)idx);
}

int64_t kodae_any_len(kodae_any a) {
    if (!a) return 0;
    if (json_value_get_type(a) == JSONArray) {
        return (int64_t)json_array_get_count(json_value_get_array(a));
    }
    if (json_value_get_type(a) == JSONObject) {
        return (int64_t)json_object_get_count(json_value_get_object(a));
    }
    return 0;
}

int64_t kodae_any_int(kodae_any a) {
    if (!a) return 0;
    return (int64_t)json_value_get_number(a);
}

double kodae_any_float(kodae_any a) {
    if (!a) return 0.0;
    return json_value_get_number(a);
}

kodae_str kodae_any_str(kodae_any a) {
    if (!a) return (kodae_str){NULL, 0};
    const char* s = json_value_get_string(a);
    if (!s) return (kodae_str){NULL, 0};
    return kodae_str_new(s);
}

bool kodae_any_bool(kodae_any a) {
    if (!a) return false;
    return json_value_get_boolean(a) != 0;
}
